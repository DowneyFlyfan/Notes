# @的作用
1.  装饰器: 在不改变函数的前提下，添加额外功能  
    def my_decorator(func):  
        def wrapper():  
            print("Something before the function.")  
            func()  
            print("Something after the function.")  
        return wrapper  
      
    @my_decorator  
    def say_hello():  
        print("Hello!")  
    say_hello()  
      
    
2.  类方法装饰器  
    class MyClass:  
        class_attribute = "This is class attribute"  
      
        @staticmethod  
        def static_method(): # 在**不传入实例 (self)** 的情况下调用方法，  
            print("This is a static method.")  
      
        @classmethod    # 类方法，自动传入类作为第一个参数, 在不传入实例的情况下调用方法和**类的属性**  
        def class_method(cls):  
            print(cls.classattribute)  
      
        @property #   # 属性，将方法定义为属性  
        def some_property(self):  
            return "This is a property"  
      
    my_object = MyClass()  
    MyClass.static_method()  
    MyClass.class_method()   
    print(my_object.some_property)  
      
    
3.  装饰器链  
    \# 这个函数先被two装饰，再被one装饰  
    @decorator_one  
    @decorator_two  
    def my_function():  
        pass  
      
    
4.  矩阵乘法符号 (python >= 3.5)  
    import numpy as np  
      
    A = np.array(\[\[1, 2], \[3, 4]])  
    B = np.array(\[\[5, 6], \[7, 8]])  
      
    result = A @ B  
    print(result)  
      
    
5.  装饰器参数  
    def repeat(times):  
        def decorator(func):  
            def wrapper(\*args, \*\*kwargs):  
                for _ in range(times):  
                    func(\*args, \*\*kwargs)  
            return wrapper  
        return decorator  
      
    \# 使得say_hello函数在调用时执行3次  
    @repeat(3)  
    def say_hello():  
        print("Hello!")  
      
    say_hello()


# \*的作用
1.  Multiplication & Repetition  
    a = 2\*3  
    print(f"a= {a}")  
      
    String = "Hello"\*2  
    print(f"String is {String}")  
      
    list = \[1,2,3]\*3  
    print(f"List is {list}")  
      
    
2.  Unpacking & Collecting Argument of Lists  
    def sum_up(a,b,c):  
        return a + b + c  
      
    arg = \[1,2,3]  
    result = sum_up(\*arg)  
    print(f"Result is {result}")  
      
    def sum_all(\*args):  
        return sum(args)  
      
    print(f"Sum is {sum_all(1,2,3,4)}")  
      
    
3.  Unpacking & Collecting Arguments of Dictionaries  
      
    def greet(name, greeting):  
        return f"{greeting}: {name}"  
      
    kargs = {name: "Alice", greeting: "Hello"}  
    print(greet(\*\*kargs))  
      
    def greeting(\*\*kargs):  
        for key, value in kargs.items():  
            print(f"{key}: {value}")  
      
    greeting(Name = "Eric", Age = 30, Job = "Engineer")

    函数阐述解包后的参数是 迭代器\[传入列表] / 字典 \[ 传入字典 ]


# _的作用
1.  _var (前面一个下划线):  
    表明这变量是 **内部的，受保护的，不建议从外部直接访问的  
    **  
    
2.  __var(前面两个下划线):  
    触发名称重整机制，防止后面的方法或者类与其名称冲突  
      
    class MyClass:  
        def __init__(self):  
            self.__hidden = 42  # 在访问时，实际上看到的属性是 _MyClass__hidden  
      
    
3.  __var__(前后双下划线):  
    python的类中的特殊/魔法方法  
      
    
4.  var_(后单下划线)：  
    防止名称冲突 ( 同两个下划线 )  
    或者是pytorch中的 **inplace操作**，不增加新的内存，提高效率  
      
    
5.  单个下划线作为变量名, 代替不重要的，可以忽略的变量:  
    for _ in range(10):  
        do_sth()
